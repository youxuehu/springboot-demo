# 停止所有的容器
docker stop $(docker ps -a -q)

# 删除所有的容器
docker rm $(docker ps -a -q)

# 查看所有镜像
docker images

# 删除指定id的镜像
docker rmi <image id>

# 删除全部的images
docker rmi $(docker images -q)
# 强制删除
docker rmi -f $(docker images -q)

# 删除untagged images
docker rmi $(docker images | grep "^" | awk "{print $3}")
